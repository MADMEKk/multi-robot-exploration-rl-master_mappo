# STANDARD COMMANDS

# Start the environment (default)
ros2 launch start_rl_environment main.launch.py map_number:=1 robot_number:=2

# Start the environment with all arguments
ros2 launch start_rl_environment main.launch.py map_number:=1 robot_number:=2 headless:=false fast_training:=false

# Training with MAPPO (default)
ros2 launch start_reinforcement_learning start_learning.launch.py map_number:=1 robot_number:=2 algorithm:=mappo

# Training with MADDPG
ros2 launch start_reinforcement_learning start_learning.launch.py map_number:=1 robot_number:=2 algorithm:=maddpg

# Testing with MAPPO (default)
ros2 launch start_reinforcement_learning evaluate.launch.py map_number:=1 robot_number:=2 algorithm:=mappo

# Testing with MADDPG
ros2 launch start_reinforcement_learning evaluate.launch.py map_number:=1 robot_number:=2 algorithm:=maddpg

# Algorithm comparison
ros2 launch start_reinforcement_learning compare_algorithms.launch.py map_number:=1 robot_number:=2 n_episodes:=100


# FAST TRAINING COMMANDS

# Start the environment in headless mode with fast training
ros2 launch start_rl_environment main.launch.py map_number:=1 robot_number:=2 headless:=true fast_training:=true

# Using the speed_train.sh script (recommended)

# Default: Quality-optimized fast training (5x speed, good accuracy)
./speed_train.sh

# All available arguments
./speed_train.sh --map=1 --robots=2 --algorithm=mappo --headless=true --fast=true --speed=quality

# Quality-optimized fast training (5x speed, good accuracy)
./speed_train.sh --speed=quality

# Ultra-fast training (10-20x speed, reduced accuracy)
./speed_train.sh --speed=ultra

# Normal speed training (standard simulation)
./speed_train.sh --speed=normal

# Training with MADDPG algorithm
./speed_train.sh --algorithm=maddpg

# Training with GUI (not headless)
./speed_train.sh --headless=false


# AVAILABLE ARGUMENTS

# Environment arguments:
# - map_number:     Map number (default: 1)
# - robot_number:   Number of robots (default: 3)
# - headless:       Run without GUI (default: false)
# - fast_training:  Enable optimized physics (default: false)

# Learning arguments:
# - map_number:     Map number (default: 1)
# - robot_number:   Number of robots (default: 3)
# - algorithm:      RL algorithm: mappo or maddpg (default: mappo)

# speed_train.sh arguments:
# - --map=N:        Set map number (default: 1)
# - --robots=N:     Set number of robots (default: 2)
# - --algorithm=X:  Set algorithm: mappo or maddpg (default: mappo)
# - --headless=X:   Run in headless mode: true or false (default: true)
# - --fast=X:       Enable fast training: true or false (default: true)
# - --speed=X:      Set speed level: quality, normal, or ultra (default: quality)
# - --help:         Display help message
